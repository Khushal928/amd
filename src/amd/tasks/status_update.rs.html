34/// Checks for status updates daily at 5 AM.
35pub struct StatusUpdateCheck;
36
37#[async_trait]
38impl Task for StatusUpdateCheck {
39    fn name(&self) -> &str {
40        "Status Update Check"
41    }
42
43    fn run_in(&self) -> tokio::time::Duration {
44        time_until(5, 00)
45    }
46
47    async fn run(&self, ctx: Context) -> anyhow::Result<()> {
48        status_update_check(ctx).await
49    }
50}

54struct ReportConfig {
55    time_valid_from: DateTime<chrono_tz::Tz>,
56    keywords: Vec<&'static str>,
57    special_authors: Vec<&'static str>,
58}

63async fn status_update_check(ctx: Context) -> anyhow::Result<()> {
64    let updates = get_updates(&ctx).await?;
65    let members = fetch_members().await?;
66
67    // naughty_list -> members who did not send updates
68    let (mut naughty_list, mut nice_list) = categorize_members(&members, updates);
69    update_streaks_for_members(&mut naughty_list, &mut nice_list).await?;
70
71    let embed = generate_embed(members, naughty_list).await?;
72    let msg = CreateMessage::new().embed(embed);
73
74    let status_update_channel = ChannelId::new(STATUS_UPDATE_CHANNEL_ID);
75    status_update_channel.send_message(ctx.http(), msg).await?;
76
77    Ok(())
78}

80async fn get_updates(ctx: &Context) -> anyhow::Result<Vec<Message>> {
81    let channel_ids = get_channel_ids();
82    let mut updates = Vec::new();
83
84    let get_messages_builder = GetMessages::new().limit(100);
85    for channel in channel_ids {
86        let messages = channel.messages(ctx.http(), get_messages_builder).await?;
87        let valid_updates = messages.into_iter().filter(is_valid_status_update);
88        updates.extend(valid_updates);
89    }
90
91    Ok(updates)
92}

104fn is_valid_status_update(msg: &Message) -> bool {
105    let report_config = get_report_config();
106    let content = msg.content.to_lowercase();
107
108    let is_within_timeframe = DateTime::<Utc>::from_timestamp(msg.timestamp.timestamp(), 0)
109        .expect("Valid timestamp")
110        .with_timezone(&chrono_tz::Asia::Kolkata)
111        >= report_config.time_valid_from;
112
113    let has_required_keywords = report_config
114        .keywords
115        .iter()
116        .all(|keyword| content.contains(keyword));
117    let is_special_author = report_config
118        .special_authors
119        .contains(&msg.author.id.to_string().as_str());
120    let is_valid_content =
121        has_required_keywords || (is_special_author && content.contains("regards"));
122
123    is_within_timeframe && is_valid_content
124}

126// TODO: Parts of this could also be removed from code like channel_ids
127fn get_report_config() -> ReportConfig {
128    let now = chrono::Utc::now().with_timezone(&chrono_tz::Asia::Kolkata);
129    let yesterday = now.date_naive() - chrono::Duration::days(1);
130    let time_valid_from = yesterday
131        .and_hms_opt(20, 0, 0)
132        .expect("Valid timestamp")
133        .and_local_timezone(chrono_tz::Asia::Kolkata)
134        .earliest()
135        .expect("Valid timezone conversion");
136
137    ReportConfig {
138        time_valid_from,
139        keywords: vec!["namah shivaya", "regards"],
140        special_authors: vec![AMAN_SHAFEEQ, CHANDRA_MOULI],
141    }
142}

144fn categorize_members(
145    members: &Vec<Member>,
146    updates: Vec<Message>,
147) -> (GroupedMember, Vec<Member>) {
148    let mut nice_list = vec![];
149    let mut naughty_list: HashMap<Option<String>, Vec<Member>> = HashMap::new();
150
151    let mut sent_updates: HashSet<String> = HashSet::new();
152
153    for message in updates.iter() {
154        sent_updates.insert(message.author.id.to_string());
155    }
156
157    for member in members {
158        if sent_updates.contains(&member.discord_id) {
159            nice_list.push(member.clone());
160        } else {
161            let track= member.track.clone();
162            naughty_list
163                .entry(track)
164                .or_insert_with(Vec::new)
165                .push(member.clone());
166        }
167    }
168
169    (naughty_list, nice_list)
170}

172async fn update_streaks_for_members(
173    naughty_list: &mut GroupedMember,
174    nice_list: &mut Vec<Member>,
175) -> anyhow::Result<()> {
176    for member in nice_list {
177        increment_streak(member).await?;
178    }
179
180    for members in naughty_list.values_mut() {
181        for member in members {
182            reset_streak(member).await?;
183        }
184    }
185
186    Ok(())
187}

189async fn generate_embed(
190    members: Vec<Member>,
191    naughty_list: GroupedMember,
192) -> anyhow::Result<CreateEmbed> {
193    let (all_time_high, all_time_high_members, current_highest, current_highest_members) =
194        get_leaderboard_stats(members).await?;
195    let mut description = String::new();
196
197    description.push_str("# Leaderboard Updates\n");
198
199    description.push_str(&format!(
200        "## All-Time High Streak: {} days\n",
201        all_time_high
202    ));
203    description.push_str(&format_members(&all_time_high_members));
204
205    description.push_str(&format!(
206        "## Current Highest Streak: {} days\n",
207        current_highest
208    ));
209    description.push_str(&format_members(&current_highest_members));
210
211    if !naughty_list.is_empty() {
212        description.push_str("# Defaulters\n");
213        description.push_str(&format_defaulters(&naughty_list));
214    }
215
216    let embed = CreateEmbed::new()
217        .title("Status Update Report")
218        .description(description)
219        .color(serenity::all::Colour::new(0xeab308));
220
221    Ok(embed)
222}

224fn format_members(members: &[Member]) -> String {
225    if members.len() <= 5 {
226        let list = members
227            .iter()
228            .map(|member| format!("- {}", member.name))
229            .collect::<Vec<_>>()
230            .join("\n");
231
232        format!("{}\n", list)
233    } else {
234        String::from("More than five members hold this record!\n")
235    }
236}
237
238fn format_defaulters(naughty_list: &GroupedMember) -> String {
239    let mut description = String::new();
240    for (track, missed_members) in naughty_list {
241        match track{
242            Some(t) => description.push_str(&format!("## Track - {}\n", t)),
243            None => description.push_str(&format!("## Unassigned"))
244        }
245
246        for member in missed_members {
247            let status = match member.streak[0].current_streak {
248                0 => ":x:",
249                -1 => ":x::x:",
250                _ => ":headstone:",
251            };
252            description.push_str(&format!("- {} | {}\n", member.name, status));
253        }
254    }
255    description.push('\n');
256    description
257}

259async fn get_leaderboard_stats(
260    members: Vec<Member>,
261) -> anyhow::Result<(i32, Vec<Member>, i32, Vec<Member>)> {
262    let streaks = fetch_streaks().await?;
263    let member_map: HashMap<i32, &Member> = members.iter().map(|m| (m.member_id, m)).collect();
264
265    let (all_time_high, all_time_high_members) = find_highest_streak(&streaks, &member_map, true);
266    let (current_highest, current_highest_members) =
267        find_highest_streak(&streaks, &member_map, false);
268
269    Ok((
270        all_time_high,
271        all_time_high_members,
272        current_highest,
273        current_highest_members,
274    ))
275}
276
277fn find_highest_streak(
278    streaks: &[StreakWithMemberId],
279    member_map: &HashMap<i32, &Member>,
280    is_all_time: bool,
281) -> (i32, Vec<Member>) {
282    let mut highest = 0;
283    let mut highest_members = Vec::new();
284
285    for streak in streaks {
286        if let Some(member) = member_map.get(&streak.member_id) {
287            let streak_value = if is_all_time {
288                streak.max_streak
289            } else {
290                streak.current_streak
291            };
292
293            match streak_value.cmp(&highest) {
294                std::cmp::Ordering::Greater => {
295                    highest = streak_value;
296                    highest_members.clear();
297                    highest_members.push((*member).clone());
298                }
299                std::cmp::Ordering::Equal => {
300                    highest_members.push((*member).clone());
301                }
302                _ => {}
303            }
304        }
305    }
306
307    (highest, highest_members)
