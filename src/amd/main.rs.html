55fn setup_tracing() -> anyhow::Result<ReloadHandle> {
56    let env = std::env::var("AMD_RUST_ENV").context("RUST_ENV was not found in the ENV")?;
57    let enable_debug_libraries_string = std::env::var("ENABLE_DEBUG_LIBRARIES")
58        .context("ENABLE_DEBUG_LIBRARIES was not found in the ENV")?;
59    let enable_debug_libraries: bool = enable_debug_libraries_string
60        .parse()
61        .context("Failed to parse ENABLE_DEBUG_LIBRARIES")?;
62    let crate_name = env!("CARGO_CRATE_NAME");
63
64    let (filter, reload_handle) = reload::Layer::new(EnvFilter::new(
65        if env == "production" && enable_debug_libraries {
66            "info".to_string()
67        } else if env == "production" && !enable_debug_libraries {
68            format!("{crate_name}=info")
69        } else if enable_debug_libraries {
70            "trace".to_string()
71        } else {
72            format!("{crate_name}=trace")
73        },
74    ));
75
76    if env != "production" {
77        let subscriber = tracing_subscriber::registry()
78            .with(filter)
79            .with(fmt::layer().pretty().with_writer(std::io::stdout))
80            .with(
81                fmt::layer()
82                    .pretty()
83                    .with_ansi(false)
84                    .with_writer(File::create("amd.log").context("Failed to create subscriber")?),
85            );
86
87        tracing::subscriber::set_global_default(subscriber).context("Failed to set subscriber")?;
88        Ok(Arc::new(RwLock::new(reload_handle)))
89    } else {
90        let subscriber = tracing_subscriber::registry().with(filter).with(
91            fmt::layer()
92                .pretty()
93                .with_ansi(false)
94                .with_writer(File::create("amd.log").context("Failed to create subscriber")?),
95        );
96
97        tracing::subscriber::set_global_default(subscriber).context("Failed to set subscriber")?;
98        Ok(Arc::new(RwLock::new(reload_handle)))
99    }
100}

102#[tokio::main]
103async fn main() -> Result<(), Error> {
104    dotenv::dotenv().ok();
105    let reload_handle = setup_tracing().context("Failed to setup tracing")?;
106
107    info!("Tracing initialized. Continuing main...");
108    let mut data = Data {
109        reaction_roles: HashMap::new(),
110        log_reload_handle: reload_handle,
111    };
112    populate_data_with_reaction_roles(&mut data);
113
114    let discord_token =
115        std::env::var("DISCORD_TOKEN").context("DISCORD_TOKEN was not found in the ENV")?;
116    let owner_id: u64 = std::env::var("OWNER_ID")
117        .context("OWNER_ID was not found in the ENV")?
118        .parse()
119        .context("Failed to parse owner_id")?;
120    let owner_user_id = UserId::from(owner_id);
121
122    let framework = Framework::builder()
123        .options(FrameworkOptions {
124            commands: commands::get_commands(),
125            event_handler: |ctx, event, framework, data| {
126                Box::pin(event_handler(ctx, event, framework, data))
127            },
128            prefix_options: PrefixFrameworkOptions {
129                prefix: Some(String::from("$")),
130                ..Default::default()
131            },
132            owners: HashSet::from([owner_user_id]),
133            ..Default::default()
134        })
135        .setup(|ctx, _ready, framework| {
136            Box::pin(async move {
137                poise::builtins::register_globally(ctx, &framework.options().commands).await?;
138                scheduler::run_scheduler(ctx.clone()).await;
139                Ok(data)
140            })
141        })
142        .build();
143
144    let mut client = serenity::client::ClientBuilder::new(
145        discord_token,
146        GatewayIntents::non_privileged() | GatewayIntents::MESSAGE_CONTENT,
147    )
148    .framework(framework)
149    .await
150    .context("Failed to create the Serenity client")?;
151
152    client
153        .start()
154        .await
155        .context("Failed to start the Serenity client")?;
156
157    info!("Starting amD...");
158
159    Ok(())
160}

162async fn event_handler(
163    ctx: &SerenityContext,
164    event: &FullEvent,
165    _framework: poise::FrameworkContext<'_, Data, Error>,
166    data: &Data,
167) -> Result<(), Error> {
168    match event {
169        FullEvent::ReactionAdd { add_reaction } => {
170            handle_reaction(ctx, add_reaction, data, true).await;
171        }
172        FullEvent::ReactionRemove { removed_reaction } => {
173            handle_reaction(ctx, removed_reaction, data, false).await;
174        }
175        _ => {}
176    }
177
178    Ok(())
179}
