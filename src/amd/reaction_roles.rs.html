50pub async fn handle_reaction(
51    ctx: &SerenityContext,
52    reaction: &Reaction,
53    data: &Data,
54    is_add: bool,
55) {
56    if !is_relevant_reaction(reaction.message_id, &reaction.emoji, data) {
57        return;
58    }
59
60    debug!("Handling {:?} from {:?}.", reaction.emoji, reaction.user_id);
61
62    // TODO Log these errors
63    let Some(guild_id) = reaction.guild_id else {
64        return;
65    };
66    let Some(user_id) = reaction.user_id else {
67        return;
68    };
69    let Ok(member) = guild_id.member(ctx, user_id).await else {
70        return;
71    };
72    let Some(role_id) = data.reaction_roles.get(&reaction.emoji) else {
73        return;
74    };
75
76    let result = if is_add {
77        member.add_role(&ctx.http, *role_id).await
78    } else {
79        member.remove_role(&ctx.http, *role_id).await
80    };
81
82    if let Err(e) = result {
83        error!(
84            "Could not handle {:?} from {:?}. Error: {}",
85            reaction.emoji, reaction.user_id, e
86        );
87    }
88}

90fn is_relevant_reaction(message_id: MessageId, emoji: &ReactionType, data: &Data) -> bool {
91    message_id == MessageId::new(ROLES_MESSAGE_ID) && data.reaction_roles.contains_key(emoji)
